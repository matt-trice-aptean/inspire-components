  /* Track */
  --slider-track-bg:              var(--color-text-disabled);   /* → #cfcece */
  --slider-track-bg-disabled:     var(--color-border-default);  /* → #e7e6e6 */
  --slider-track-height:          4px;
  --slider-track-radius:          2px;

  /* Range */
  --slider-range-bg:              var(--color-text-default);    /* → #3f3e3d */
  --slider-range-bg-disabled:     var(--color-text-disabled);   /* → #cfcece */

  /* Handle — 16px solid circle, 4px inner stroke, ADK Base drop shadow */
  --slider-handle-size:           16px;
  --slider-handle-bg:             var(--color-text-default);      /* → #3f3e3d */
  --slider-handle-stroke:              var(--color-surface-default);   /* inner: #ffffff */
  --slider-handle-stroke-width:        4px;
  --slider-handle-stroke-width-active: 3px;                          /* narrows on active */
  --slider-handle-shadow:         0 1px 4px rgba(0, 0, 0, 0.18); /* ADK Base */

  /* Handle — hover: ADK 70 / Slider/Handle/Hover */
  --slider-handle-bg-hover:       var(--color-border-emphasis);    /* → #575655 */

  /* Handle — focus: ADK 70 fill, 2px blue ring at edge */
  --slider-handle-bg-focus:       var(--color-border-emphasis);    /* Slider/Handle/Focus → ADK 70 → #575655 */
  --slider-handle-ring-focus:     #1a88b7;                          /* Border/Focus (SHARED SEMANTIC) */
  --slider-handle-ring-width:     2px;
  --slider-handle-ring-offset:    0px;                              /* on edge, not outside */

  /* Handle — active: ADK 70 — Slider/Handle/Active */
  --slider-handle-bg-active:      var(--color-border-emphasis);  /* → #575655 */

  /* Handle — disabled: muted gray, no shadow */
  --slider-handle-bg-disabled:         var(--color-text-disabled);    /* → #cfcece */
  --slider-handle-stroke-disabled:     var(--color-surface-disabled); /* → #f2f2f2 */

  /* Label — Inter Tight (GermanedgeSansCn substitute) */
  --slider-label-text:            var(--color-text-default);    /* → #3f3e3d */
  --slider-label-text-disabled:   var(--color-text-disabled);   /* → #cfcece */
  --slider-label-font-size:       12px;
  --slider-label-line-height:     16px;
}
