/* Slider component variables
   Tier 2 exception — unique visual roles with no semantic equivalent.
   All vars reference semantic tokens. Raw hex only in dark override.
   Figma collection: COMPONENTS · Path prefix: Slider/ */
:root {
  /* Track */
  --slider-track-bg:                   var(--color-text-disabled);    /* → #cfcece */
  --slider-track-bg-disabled:          var(--color-border-default);   /* → #e7e6e6 */
  --slider-track-height:               4px;
  --slider-track-radius:               2px;

  /* Range */
  --slider-range-bg:                   var(--color-text-default);     /* → #3f3e3d */
  --slider-range-bg-disabled:          var(--color-text-disabled);    /* → #cfcece */

  /* Handle */
  --slider-handle-size:                16px;
  --slider-handle-bg:                  var(--color-text-default);
  --slider-handle-stroke:              var(--color-surface-default);  /* inner ring: #ffffff */
  --slider-handle-stroke-width:        4px;
  --slider-handle-stroke-width-active: 3px;                           /* narrows on drag */
  --slider-handle-shadow:              0 1px 4px rgba(0,0,0,0.18);   /* Elevation/ADK Base */
  --slider-handle-bg-hover:            var(--color-border-emphasis);  /* → #575655 */
  --slider-handle-bg-focus:            var(--color-border-emphasis);
  --slider-handle-ring-focus:          #1a88b7;                       /* Border/Focus (SHARED SEMANTIC) */
  --slider-handle-ring-width:          2px;
  --slider-handle-ring-offset:         0px;
  --slider-handle-bg-active:           var(--color-border-emphasis);
  --slider-handle-bg-disabled:         var(--color-text-disabled);    /* → #cfcece */
  --slider-handle-stroke-disabled:     var(--color-surface-disabled); /* → #f2f2f2 */
  /* cursor: not-allowed on .dx-state-disabled root — pointer-events: none omitted
     so the cursor is visible; JS initSlider() guard prevents interaction */

  /* Label */
  --slider-label-text:                 var(--color-text-default);
  --slider-label-text-disabled:        var(--color-text-disabled);
  --slider-label-font-size:            12px;
  --slider-label-line-height:          16px;
  --slider-label-gap:                  0px;
}
